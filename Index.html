        function updateNavigation() {

            // Disable/Enable buttons

            prevBtn.disabled = currentSlideIndex === -1;

            nextBtn.disabled = currentSlideIndex === presentationData.slides.length - 1;


            // Generate the exact dots logic from React

            let dotsHtml = '';

            

            // First dot (Cover indicator)

            dotsHtml += `<div class="w-2.5 h-2.5 rounded-full transition-all duration-500 ${currentSlideIndex === -1 ? 'bg-indigo-600 w-6' : 'bg-slate-300'}"></div>`;

            

            // Slide dots

            presentationData.slides.forEach((_, i) => {

                dotsHtml += `<div class="w-2.5 h-2.5 rounded-full transition-all duration-500 ${currentSlideIndex === i ? 'bg-indigo-600 w-6' : 'bg-slate-300'}"></div>`;

            });

            

            paginationDots.innerHTML = dotsHtml;

        }

        // Gemini API Logic

        const fetchWithRetry = async (url, options, retries = 5, delay = 1000) => {

            try {

                const res = await fetch(url, options);

                if (!res.ok) throw new Error(`HTTP error! status: ${res.status}`);

                return await res.json();

            } catch (error) {

                if (retries > 0) {

                    await new Promise(resolve => setTimeout(resolve, delay));

                    return fetchWithRetry(url, options, retries - 1, delay * 2);

                }

                throw error;

            }

        };
